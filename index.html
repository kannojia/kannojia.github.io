json-query

A tool which allows to query JSON objects using SQL like query statements

Introduction

json-query is a micro tool which you can use to write to query JSON object using SQL like query statements. This tool is still in its infancy.

You can get the script here jsonQuery.

Usage

Include the script in your page

<script src="path/to/script.js"></script>
<!-- Rest of the HTML -->

and use it as follows:

var testJSON = [
  ['name', 'owner', 'species', 'sex', 'birth', 'death'],
  ["Fluffy", "Harold", "cat", "f", "1993-02-04", null], 
  ["Claws", "Gwen", "cat", "m", "1994-03-17", null], 
  ["Buffy", "Harold", "dog", "f", "1989-05-13", null], 
  ["Fang", "Benny", "dog", "m", "1990-08-27", null], 
  ["Bowser", "Diane", "dog", "m", "1979-08-31", "1995-07-29"], 
  ["Chirpy", "Gwen", "bird", "f", "1998-09-11", null], 
  ["Whistler", "Gwen", "bird", null, "1997-12-09", null], 
  ["Slim", "Benny", "snake", "m", "1996-04-29", null], 
  ["Puffball", "Diane", "hamster", "f", "1999-03-30", null]
];

var results = new jsonQuery(testJSON)
    .Select(['name', 'owner', 'species', 'sex'])
    .OrderBy('name', 'ASC')
    .Execute();

console.log(results);

Other examples: 

var testJSON = [
  ['name', 'owner', 'species', 'sex', 'birth', 'death'],
  ["Fluffy", "Harold", "cat", "f", "1993-02-04", null], 
  ["Claws", "Gwen", "cat", "m", "1994-03-17", null], 
  ["Buffy", "Harold", "dog", "f", "1989-05-13", null], 
  ["Fang", "Benny", "dog", "m", "1990-08-27", null], 
  ["Bowser", "Diane", "dog", "m", "1979-08-31", "1995-07-29"], 
  ["Chirpy", "Gwen", "bird", "f", "1998-09-11", null], 
  ["Whistler", "Gwen", "bird", null, "1997-12-09", null], 
  ["Slim", "Benny", "snake", "m", "1996-04-29", null], 
  ["Puffball", "Diane", "hamster", "f", "1999-03-30", null]
];

var results = new jsonQuery(testJSON)
    .Select(['name', 'owner', 'species'])
    .Where([
        ['sex', '=', 'f']
    ])
    .OrderBy('name')
    .Execute();

console.log(results);

// [
//    [ 'Buffy', 'Harold', 'dog', 'f' ],
//    [ 'Chirpy', 'Gwen', 'bird', 'f' ],
//    [ 'Fluffy', 'Harold', 'cat', 'f' ] 
//    [ 'Puffball', 'Diane', 'hamster', 'f' ],
//    [ 'Whistler', 'Gwen', 'bird', null ]
// ]


You can even write compound where clause by placing 'AND', 'OR' before the clause itself and the following clauses will be evaluated as that. To form complex conditionals you can nest conditions in similar fashion.

var results = new jsonQuery(testJSON)
    .Select(['name', 'owner', 'species'])
    .Where([
        ['OR',
            ['AND',
                ['sex', '=', 'f'],
                ['death', '!=', null]
            ],
            ['species', 'IN', ['cat', 'dog']]
        ]
    ])
    .Execute();

console.log(results);
